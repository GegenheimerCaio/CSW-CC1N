             import socket

             def abrir_porta_tcp(host, porta):
                 servidor = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
                 try:
                     servidor.bind((host, porta))
                     servidor.listen(5)
                     print(f"Porta {porta} aberta, aguardando conexões...")
                     while True:
                         conexao, endereco = servidor.accept()
                         print(f"Conexão recebida de {endereco}")
                         conexao.sendall(b'Bem-vindo ao servidor!')
                         conexao.close()
                 except Exception as e:
                     print(f"Erro ao abrir a porta: {e}")
                 finally:
                     servidor.close()
                     print(f"Porta {porta} fechada.")
        
            # Exemplo de uso
            host = '127.0.0.1'  # Endereço local
            porta = 65432       # Porta TCP
            abrir_porta_tcp(host, porta)

            # Função para calcular o fatorial
            def fatorial(n):
                if n == 0 or n == 1:
                    return 1
                else:
                    return n * fatorial(n - 1)

            # Solicita ao usuário um número
            numero = int(input("Digite um número para calcular o fatorial: "))

            # Calcula e exibe o fatorial
            resultado = fatorial(numero)
            print(f"O fatorial de {numero} é {resultado}.")
